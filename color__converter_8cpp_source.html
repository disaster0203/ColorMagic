138 color_space::hsi* color_manipulation::color_converter::rgb_deep_to_hsi(color_space::rgb_deepcolor* color)
139 {
140  float min = std::fmin(std::fmin(color->red(), color->green()), color->blue());
141  float max = std::fmax(std::fmax(color->red(), color->green()), color->blue());
142  if (max == min) // red = green = blue
143  {
144  return new color_space::hsi(0.f, 0.f, min, color->alpha(), color->get_rgb_color_space());
145  }
146  else
147  {
148  float delta = max - min;
149  float hue = hue_from_rgb_helper(color->red(), color->green(), color->blue(), max, min, delta);
150  float intensity = (color->red() + color->green() + color->blue()) / 3.f;
151  float saturation = 1.f - (min / intensity);
152  return new color_space::hsi(hue, saturation, intensity, color->alpha(), color->get_rgb_color_space());
153  }
