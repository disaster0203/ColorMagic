647  static float dissolve_func(float s, float d, float alpha_diff)
648  {
649  float rand_val = rand() % 101; // generate a random number in the range 0 - 100
650  float threshold = 25.f; // threshold of random number that defines whether to return s or d
651  if (alpha_diff > 0.f) // source alpha is greater
652  {
653  return (rand_val >= threshold) ? s : d; // return s in 3/4 of the cases
654  }
655  else // destination alpha is greater
656  {
657  return (rand_val >= threshold) ? d : s; // return d in 3/4 of the cases
658  }
659  return 0;
660  }
