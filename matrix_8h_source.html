363  T determinante()
364  {
365  // First check special cases to shorten calculation for this cases:
366  // -------------------------------------------------------------------------
367 
368  // 1. Is the matrix quadratic (NxN)? If not calculating the determinante is impossible -> return 0.
369  if (m_rows != m_columns) return 0;
370 
371  // 2. Do we have a 1x1 matrix (just one value)? -> Determinante = value
372  if (m_rows == 1) return m_values[0][0];
373 
374  // 3. Is one of the rows filled with 0? -> Determinante = 0
375  bool row_is_null = false;
376  for (int row = 0; row < m_rows; ++row)
377  {
378  row_is_null = is_row_null(row);
379  if (row_is_null) break;
380  }
381  if (row_is_null) return 0;
382 
383  // 4. Is one of the columns filled with 0? -> Determinante = 0
384  bool col_is_null = false;
385  for (int col = 0; col < m_rows; ++col)
386  {
387  col_is_null = is_column_null(col);
388  if (col_is_null) break;
389  }
390  if (col_is_null) return 0;
391 
392  // 5. Are two rows equal? -> Determinante = 0
393  bool rows_equal = false;
394  for (int i = 0; i < m_rows; ++i)
395  {
396  if (rows_equal) break;
397  for (int j = i + 1; j < m_rows; ++j)
398  {
399  rows_equal = are_rows_equal(i, j);
400  if (rows_equal) break;
401  }
402  }
403  if (rows_equal) return 0;
404 
405  // 6. Are two columns equal? -> Determinante = 0
406  bool cols_equal = false;
407  for (int i = 0; i < m_rows; ++i)
408  {
409  if (cols_equal) break;
410  for (int j = i + 1; j < m_rows; ++j)
411  {
412  cols_equal = are_columns_equal(i, j);
413  if (cols_equal) break;
414  }
415  }
416  if (cols_equal) return 0;
417 
418  // I could also check if one row/column is a multiple of another row/column
419  // but I will leave this. In this case the determinante would also be 0.
420 
421  // -------------------------------------------------------------------------
422 
423  // All checks failed. So we do have to calculate the determinante...
424  return calculate_determinante(m_values, m_rows);
425  }
