112 color_space::color_base * color_manipulation::chromatic_adaptation::bradford_adaptation(color_space::color_base * color, color_space::white_point * target_white_point)
113 {
114  if (color->get_rgb_color_space()->get_white_point() == target_white_point)
115  {
116  return color;
117  }
118 
119  // Convert to XYZ space and transform using bradford matrix (incl. normalization)
120  auto tmp_color = color_manipulation::color_converter::to_xyz(color);
121  auto rgb_comp = m_bradford * tmp_color->get_component_vector();
122  rgb_comp[0] /= color->get_component_vector()[1];
123  rgb_comp[1] /= color->get_component_vector()[1];
124  rgb_comp[2] /= color->get_component_vector()[1];
125 
126  // Create scaled white point vectors (incl. normalization)
127  std::vector<float> source_wp = std::vector<float>();
128  source_wp.push_back(color->get_rgb_color_space()->get_white_point()->get_tristimulus()[0] / color->get_rgb_color_space()->get_white_point()->get_tristimulus()[1]);
129  source_wp.push_back(color->get_rgb_color_space()->get_white_point()->get_tristimulus()[1] / color->get_rgb_color_space()->get_white_point()->get_tristimulus()[1]);
130  source_wp.push_back(color->get_rgb_color_space()->get_white_point()->get_tristimulus()[2] / color->get_rgb_color_space()->get_white_point()->get_tristimulus()[1]);
131 
132  std::vector<float> dest_wp = std::vector<float>();
133  dest_wp.push_back(target_white_point->get_tristimulus()[0] / target_white_point->get_tristimulus()[1]);
134  dest_wp.push_back(target_white_point->get_tristimulus()[1] / target_white_point->get_tristimulus()[1]);
135  dest_wp.push_back(target_white_point->get_tristimulus()[2] / target_white_point->get_tristimulus()[1]);
136 
137  auto scaled_source_wp = m_bradford * source_wp;
138  auto scaled_dest_wp = m_bradford * dest_wp;
139 
140  // Adapt color (incl. undo of normalization)
141  auto p = powf(scaled_source_wp[2] / scaled_dest_wp[2], 0.0834f);
142  std::vector<float> tmp_rgb_comp;
143  tmp_rgb_comp.push_back((scaled_dest_wp[0] * (rgb_comp[0] / scaled_source_wp[0])) * color->get_component_vector()[1]);
144  tmp_rgb_comp.push_back((scaled_dest_wp[1] * (rgb_comp[1] / scaled_source_wp[1])) * color->get_component_vector()[1]);
145 
146  float div = rgb_comp[2] / scaled_source_wp[2];
147  tmp_rgb_comp.push_back((scaled_dest_wp[2] * powf(fabsf(div), p)) * color->get_component_vector()[1]); // avoid overflow of float
148  if (div < 0.f) tmp_rgb_comp[2] *= -1.f;
149 
150  auto transformed_components = m_inverted_bradford * tmp_rgb_comp;
151  auto rgb_def = new color_space::rgb_color_space_definition(*color->get_rgb_color_space());
152  rgb_def->set_white_point(target_white_point);
153  auto tmp_trans_color = new color_space::xyz(transformed_components[0], transformed_components[1], transformed_components[2], tmp_color->alpha(), rgb_def);
154 
155  // Convert transformed color back to input color space
156  return color_manipulation::color_converter::convertTo(tmp_trans_color, color->get_color_type());
157 }

364 color_space::color_base * color_manipulation::chromatic_adaptation::do_adaption(color_space::color_base * color, color_space::white_point * target_white_point, matrix<float> mat, matrix<float> inverted_mat)

367  auto tmp_color = color_manipulation::color_converter::to_xyz(color);
368

371  source_wp.push_back(color->get_rgb_color_space()->get_white_point()->get_tristimulus()[0]);
